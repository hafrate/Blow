library(tidyverse)
library(sweep)
library(timetk)
library(lubridate)
library(gridExtra)
data("bike_sales")
bike <- bike_sales %>%
  mutate(order.date = as_date(zoo::as.yearmon(order.date))) %>%
  group_by(order.date, category.secondary) %>%
  summarise(qty = sum(quantity))
bike %>%
  ggplot(aes(order.date, qty), group = category.secondary) +
  geom_line(aes(color = category.secondary))
